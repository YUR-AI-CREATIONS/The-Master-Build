from fastapi import HTTPException
import requests

def send_message(channel: str, message: str):
    token = get_token("slack")
    url = "https://slack.com/api/chat.postMessage"
    headers = {"Authorization": f"Bearer {token}"}
    data = {"channel": channel, "text": message}

"""
Simple local PQC key generator using the pyca/cryptography backend.
(For demonstration – replace with a Dilithium or SPHINCS+ library when available.)
"""

from cryptography.hazmat.primitives.asymmetric import ed25519
from cryptography.hazmat.primitives import serialization
from pathlib import Path

VAULT = Path("vault/keys")

def generate_keys():
    private_key = ed25519.Ed25519PrivateKey.generate()
    public_key = private_key.public_key()

    VAULT.mkdir(parents=True, exist_ok=True)
    (VAULT / "private.pem").write_bytes(
        private_key.private_bytes(
            encoding=serialization.Encoding.PEM,
            format=serialization.PrivateFormat.PKCS8,
            encryption_algorithm=serialization.NoEncryption(),
        )
    )
    (VAULT / "public.pem").write_bytes(
        public_key.public_bytes(
            encoding=serialization.Encoding.PEM,
            format=serialization.PublicFormat.SubjectPublicKeyInfo,
        )
    )
    return str(VAULT / "public.pem")

if __name__ == "__main__":
    print("Generated keys at", generate_keys())

"""
Sentinel key-rotation and vault integrity monitor.
Runs inside the Sentinel container or as a local service.
"""

import time
import yaml
import hashlib
from pathlib import Path
from src.core import pqc_keys

TOKENS = Path("vault/tokens.yaml")
LOG = Path("vault/rotation_log.yaml")

def file_hash(path):
    return hashlib.sha256(Path(path).read_bytes()).hexdigest()

def rotate_keys(interval_hours=168):  # 7 days default
    last_hash = ""
    while True:
        if not TOKENS.exists():
            print("No vault found, waiting...")
            time.sleep(60)
            continue

        current_hash = file_hash(TOKENS)
        if current_hash != last_hash:
            public_path = pqc_keys.generate_keys()
            log_entry = {
                "timestamp": time.time(),
                "vault_hash": current_hash,
                "new_public_key": public_path,
            }
            with open(LOG, "a") as f:
                yaml.safe_dump([log_entry], f)
            print(f"[Sentinel] Keys rotated -> {public_path}")
            last_hash = current_hash

        time.sleep(interval_hours * 3600)

if __name__ == "__main__":
    rotate_keys()
